import { generalTableSelect, highlightTableRow, readMatrixFromTable } from "./custom.js"

Inputs.button(
  html`<button style="color:black;background-color:lightgray;">Roll Background (d1000)</button>`,
  {
    value: "",
    reduce: () => {
      var matrix = readMatrixFromTable("#backgrounds-table");
      var n = matrix.length;

      // Inline 1..N ranges; no const
      var idx = generalTableSelect([1, 100, 0], Array.from({length: n}, function(_, i){ return [i+1, i+1]; }));

      highlightTableRow("#backgrounds-table", idx);

      // matrix row = [Background, Possessions, Details]
      var row = matrix[idx - 1];
      return row[0] + " — " + row[1] + ". " + row[2];
    }
  }
)

Inputs.button(
  html`<button style="color:black;background-color:lightgray;">Roll Crop Type (d8)</button>`,
  {
    value: "",
    reduce: () => {
      const matrix = readMatrixFromTable("#farmer-table");
      const n = matrix.length;

      // Build 1..n ranges so generalTableSelect can return a 1-based index
      const ranges = Array.from({length: n}, (_, i) => [i+1, i+1]);

      // Roll d8 and map to row index
      const idx = generalTableSelect([1, 8, 0], ranges);

      highlightTableRow("#farmer-table", idx);
      const row = matrix[idx - 1];
      return `d8: ${idx} → ${row[1]}`;
    }
  }
)
